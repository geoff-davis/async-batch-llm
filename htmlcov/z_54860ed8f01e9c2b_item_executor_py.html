58def _detach_traceback(exc: _E) -> _E: 

59 """Clear tracebacks along an exception's cause/context chain before it is 

60 stored on a ``WorkItemResult``. 

61 

62 A traceback pins every frame's locals — strategies, clients, raw responses — 

63 for as long as the result is held, which for a large accumulated batch of 

64 failures can retain far more memory than before ``WorkItemResult.exception`` 

65 existed. The full failure (type, message, stack) is already logged at the 

66 point it happens, so the stored exception keeps its type/message/args (enough 

67 for ``call()`` to re-raise the provider's type) but drops the frame-pinning 

68 tracebacks. A re-raise gets a fresh traceback from the raise site. 

69 """ 

70 seen: set[int] = set() 

71 cur: BaseException | None = exc 

72 while cur is not None and id(cur) not in seen: 

73 seen.add(id(cur)) 

74 cur.__traceback__ = None 

75 cur = cur.__cause__ or cur.__context__ 

76 return exc

79class ExecutorHostProtocol(Protocol[TInput, TOutput, TContext]):

704 """

709

711 exception: The exception that was raised during execution 

712 work_item: The work item being processed 

713 worker_id: ID of the worker processing this item 

714 attempt_number: Current attempt number (for logging) 

715 

716 Returns: 

717 WorkItemResult for permanent failures 

718 

719 Raises: 

720 RateLimitException: If rate limit detected (for re-queueing) 

721 Exception: If error is retryable (for retry logic to handle) 

722 """ 

723 # Try to extract token usage from failed LLM calls using robust extraction 

724 # Even if validation fails, the LLM consumed tokens 

725 failed_token_usage = self._host._extract_token_usage(exception) 

726 if failed_token_usage and failed_token_usage.get("total_tokens", 0) > 0: 

727 logger.debug( 

728 f"Extracted token usage from failed attempt for {work_item.item_id}: " 

729 f"{failed_token_usage['total_tokens']} tokens" 

730 ) 

731 

732 # Classify the error 

733 error_info = self.error_classifier.classify(exception) 

734

736 if error_info.is_rate_limit:

738 async with self._stats_lock: 

739 self._stats.rate_limit_count += 1

750 await self._handle_rate_limit(worker_id, observed_generation, error_info.suggested_wait)

752 # Re-raise the original exception to trigger retry logic 

753 # The retry loop will increment attempt and try again after cooldown 

754 raise

756 # If error is retryable, re-raise to trigger retry in _process_item_with_retries 

757 # Note: Cache invalidation is automatic because retries use different temperatures, 

758 # which creates different cache keys and bypasses any cached bad responses 

759 if error_info.is_retryable:

761 raise

764 middleware_result = await self._run_middlewares_on_error(work_item, exception)

769 error_name = type(exception).__name__ 

770 error_msg = str(exception)

1"""High-level streaming API built on the processor's first-class streaming mode. 

2 

3:func:`process_stream` yields each result as it completes; :func:`process_prompts` 

4collects them into a :class:`BatchResult`. Both accept a sync **or** async 

5iterable of prompts — bare strings get auto-generated item ids (``item_0``, 

6``item_1``, …), ``(item_id, prompt)`` pairs, or ``(item_id, prompt, context)`` 

7triples (the context flows to ``WorkItemResult.context`` and your post_processor). 

8 

9Because the processor runs workers while work is still being fed, a bounded 

10``ProcessorConfig.max_queue_size`` becomes input backpressure: the producer 

11blocks on a full work queue instead of buffering the complete source up front. 

12The result handoff queue is unbounded, so callers should consume results 

13promptly when streaming a large or unbounded source. 

14"""

41# A prompt entry is a bare prompt string, an (item_id, prompt) pair, or an

44# post_processor (handy for carrying DB ids, gold labels, routing keys, etc.).

54 which validates the shape.

69def _to_work_item( 

70 entry: PromptEntry, index: int, strategy: LLMCallStrategy[TOutput] 

71) -> LLMWorkItem[Any, TOutput, Any]: 

72 if isinstance(entry, tuple): 

73 # Cast to a variadic tuple so positional access doesn't depend on the 

74 # type checker narrowing the tuple length (which it can't do via len()). 

75 parts = cast("tuple[Any, ...]", entry) 

76 if len(parts) == 2: 

77 return LLMWorkItem(item_id=str(parts[0]), strategy=strategy, prompt=str(parts[1])) 

78 if len(parts) == 3: 

79 return LLMWorkItem( 

80 item_id=str(parts[0]), 

81 strategy=strategy, 

82 prompt=str(parts[1]), 

83 context=parts[2], 

84 ) 

85 raise TypeError( 

86 "prompt tuple must be (item_id, prompt) or (item_id, prompt, context); " 

87 f"got a tuple of length {len(parts)} at index {index}." 

88 ) 

89 return LLMWorkItem(item_id=f"item_{index}", strategy=strategy, prompt=str(entry))

92class _ProgressReporter: 

93 """Bundled ``progress=True`` reporter: tqdm when importable, else logging. 

94 

95 Implements the ``progress_callback(completed, total, current_item_id)`` 

96 contract. The tqdm bar's total follows the processor's running total, so 

97 streaming sources that keep adding work render correctly. Added in 

98 v0.19.0 (issue #100). 

99 """ 

100 

101 def __init__(self) -> None: 

102 self._bar: Any = None 

103 self._tqdm_failed = False 

104 self._closed = False 

105 self._last_logged = 0 

106 self._max_completed = 0 

107 self._max_total = 0 

108 # Sync progress callbacks are dispatched via asyncio.to_thread, so 

109 # concurrent items invoke __call__ from DIFFERENT threads — the whole 

110 # update must be atomic or two callbacks can race the lazy bar 

111 # creation and each build one (seen as a flaky double bar in CI). 

112 self._mutex = threading.Lock() 

113 

114 def __call__(self, completed: int, total: int, current_item_id: str) -> None: 

115 with self._mutex: 

116 if self._closed: 

117 # A cancelled asyncio.to_thread callback doesn't stop its 

118 # thread — a straggler can land after close(). Creating a 

119 # fresh bar then would leak a second bar; drop the update. 

120 return 

121 # Updates can also arrive out of order — keep the display monotonic. 

122 self._max_completed = max(self._max_completed, completed) 

123 self._max_total = max(self._max_total, total) 

124 completed = self._max_completed 

125 total = self._max_total 

126 if not self._tqdm_failed: 

127 if self._bar is None: 

128 try: 

129 from tqdm.auto import tqdm # type: ignore[import-not-found] 

130 

131 self._bar = tqdm(total=total, unit="item", desc="batch") 

132 except ImportError: 

133 self._tqdm_failed = True 

134 logger.info( 

135 "progress=True: tqdm is not installed " 

136 "(pip install 'async-batch-llm[progress]'); " 

137 "falling back to interval logging." 

138 ) 

139 if self._bar is not None: 

140 if self._bar.total != total: 

141 self._bar.total = total 

142 self._bar.n = completed 

143 self._bar.refresh() 

144 return 

145 # Logging fallback: every 10 items and on completion. 

146 if completed - self._last_logged >= 10 or completed >= total: 

147 self._last_logged = completed 

148 logger.info("progress: %d/%d items completed", completed, total) 

149 

150 def close(self) -> None: 

151 with self._mutex: 

152 self._closed = True 

153 if self._bar is not None: 

154 self._bar.close() 

155 self._bar = None

211async def _process_stream_impl( 

212 strategy: LLMCallStrategy[TOutput], 

213 prompts: PromptSource, 

214 *, 

215 config: ProcessorConfig | None = None, 

216 concurrency: int | None = None, 

217 progress: bool | ProgressCallbackFunc = False, 

218 artifact_store: ArtifactStore | None = None, 

219 resume: ResumePolicy = ResumePolicy.NONE, 

220 termination_out: list[BatchTermination] | None = None, 

221 **processor_kwargs: Any, 

222) -> AsyncIterator[WorkItemResult[TOutput, Any]]: 

223 """Yield each :class:`WorkItemResult` as it completes, in completion order. 

224 

225 Args: 

226 strategy: The LLM call strategy to run for every prompt. 

227 prompts: Sync or async iterable of bare prompt strings (ids 

228 auto-generated), ``(item_id, prompt)`` pairs, and/or 

229 ``(item_id, prompt, context)`` triples (context → result.context). 

230 config: Processor configuration. Defaults to ``ProcessorConfig()``. Set 

231 ``max_queue_size`` to bound memory for very large inputs. 

232 **processor_kwargs: Forwarded to ``ParallelBatchProcessor`` (observers, 

233 middlewares, ``error_classifier``, ``progress_callback``, …). 

234 

235 Yields: 

236 Results (success *and* failure) in the order items finish — generally 

237 NOT the order submitted. A producer error (e.g. a failing async input) 

238 propagates to the consumer after already-queued results drain; breaking 

239 out of the loop early cancels the producer and tears down the workers. 

240 """ 

241 config = _apply_concurrency_shorthand(config, concurrency) 

242 reporter = _resolve_progress(progress, processor_kwargs) 

243 processor = ParallelBatchProcessor( 

244 config=config or ProcessorConfig(), 

245 artifact_store=artifact_store, 

246 resume=resume, 

247 **processor_kwargs, 

248 ) 

249 feed_error: list[BaseException] = [] 

250 

251 async def _feed() -> None: 

252 try: 

253 index = 0 

254 async for entry in _aiter(prompts): 

255 await processor.add_work(_to_work_item(entry, index, strategy)) # backpressure here 

256 index += 1 

257 except asyncio.CancelledError: 

258 if not processor.aborted: 

259 raise # consumer broke out early; cleanup handles teardown 

260 except BatchAdmissionStopped: 

261 pass # controlled batch timeout/fail-fast stopped source admission 

262 except BaseException as exc: # noqa: BLE001 - surfaced to the consumer below 

263 feed_error.append(exc) 

264 finally: 

265 # Normal completion, controlled abort, or producer error: close the 

266 # accepted stream so every queued item receives a terminal result. 

267 if not processor._finished: 267 ↛ exitline 267 didn't return from function '_feed' because the condition on line 267 was always true

268 await processor.finish() 

269 

270 try: 

271 async with processor: # __aexit__ -> cleanup() cancels workers/finalize 

272 processor.start() 

273 producer = asyncio.create_task(_feed()) 

274 

275 async def _stop_producer_on_abort() -> None: 

276 await processor.wait_for_abort() 

277 if not producer.done(): 

278 producer.cancel() 

279 

280 abort_watcher = asyncio.create_task(_stop_producer_on_abort()) 

281 try: 

282 async for result in processor.results(): 

283 yield result 

284 finally: 

285 abort_watcher.cancel() 

286 if not producer.done(): 

287 producer.cancel() 

288 with contextlib.suppress(asyncio.CancelledError): 

289 await producer 

290 await asyncio.gather(abort_watcher, return_exceptions=True) 

291 finally: 

292 # After the async-with: cleanup() has drained in-flight progress 

293 # callbacks, so closing here can't race a late update re-creating 

294 # the bar (the reporter also drops post-close updates as a backstop). 

295 if reporter is not None: 

296 reporter.close() 

297 

298 if termination_out is not None: 

299 termination_out.append(processor.termination) 

300 

301 # Reached only when results() ended normally (end-of-stream). If the 

302 # producer failed, surface its exception now (after draining results). 

303 if feed_error: 

304 raise feed_error[0]

342async def process_prompts( 

343 strategy: LLMCallStrategy[TOutput], 

344 prompts: PromptSource, 

345 *, 

346 config: ProcessorConfig | None = None, 

347 concurrency: int | None = None, 

348 progress: bool | ProgressCallbackFunc = False, 

349 preserve_order: bool = False, 

350 artifact_store: ArtifactStore | None = None, 

351 resume: ResumePolicy = ResumePolicy.NONE, 

352 **processor_kwargs: Any, 

353) -> BatchResult[TOutput, Any]: 

354 """Run ``strategy`` over ``prompts`` and collect every result. 

355 

356 The one-liner entry point — drains :func:`process_stream` into a 

357 :class:`BatchResult` (whose ``results`` are in completion order). For 

358 lazy, bounded-input processing of very large inputs, use 

359 :func:`process_stream` directly with a positive 

360 ``config.max_queue_size`` and consume results promptly. 

361 

362 ``concurrency=N`` is shorthand for ``ProcessorConfig(concurrency=N)`` — 

363 the single knob that coherently sizes workers, provider admission, and 

364 built-in model connection pools (v0.19.0). 

365 

366 ``progress=True`` renders a tqdm bar when tqdm is installed 

367 (``pip install 'async-batch-llm[progress]'``) and falls back to interval 

368 logging otherwise; pass a callable ``(completed, total, current_item_id)`` 

369 for a custom reporter. Works for both collected and streamed runs 

370 (v0.19.0). 

371 """ 

372 termination: list[BatchTermination] = [] 

373 started = time.monotonic() 

374 results = [ 

375 result 

376 async for result in _process_stream_impl( 

377 strategy, 

378 prompts, 

379 config=config, 

380 concurrency=concurrency, 

381 progress=progress, 

382 artifact_store=artifact_store, 

383 resume=resume, 

384 termination_out=termination, 

385 **processor_kwargs, 

386 ) 

387 ] 

388 batch = BatchResult( 

389 results=results, 

390 termination=termination[0] if termination else BatchTermination(), 

391 wall_time_seconds=time.monotonic() - started, 

392 ) 

393 return batch.in_input_order() if preserve_order else batch

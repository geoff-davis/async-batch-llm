1"""Queue-less rate-limited gateway: many callers, one shared cooldown. 

2 

3A long-lived service object for a web app's request path. Many concurrent 

4callers each :meth:`submit` one prompt; an :class:`asyncio.Semaphore` caps 

5global concurrency against the provider, and a single shared 

6``RateLimitCoordinator`` (inside the host) gives one coordinated cooldown — one 

7caller's 429 briefly throttles everyone, then slow-starts. 

8 

9There is no queue, no worker pool, no background dispatcher, and no per-request 

10Future demux: each caller's coroutine runs :meth:`ItemExecutor.execute` 

11directly under the semaphore and returns its own result. A cancelled caller 

12(client disconnect) simply releases its slot — nothing is orphaned. 

13 

14For load, two opt-in knobs bound the request path: ``max_pending`` (an admission 

15cap that rejects instantly instead of growing an unbounded waiter list) and 

16``submit_timeout`` (a per-caller latency budget). Both are off by default. 

17 

18 from contextlib import asynccontextmanager 

19 from async_batch_llm import OpenAIModel, OpenAIStrategy, ProcessorConfig 

20 from async_batch_llm.gateway import LLMGateway

24 strategy = OpenAIStrategy(OpenAIModel.from_api_key("gpt-4o-mini")) 

25 async with LLMGateway(strategy, config=ProcessorConfig(max_workers=5)) as gw: 

26 app.state.llm = gw 

27 yield 

28 

29 # in a handler: summary = await request.app.state.llm.submit(prompt) 

30"""

47class LLMGateway(Generic[TOutput]): 

48 """A shared, rate-limited entry point for single LLM calls from many callers. 

49 

50 Create one at startup and call :meth:`submit` from any number of concurrent 

51 handlers. ``config.max_workers`` is the global concurrency budget. The 

52 strategy is shared, so use a stateless strategy or one whose ``prepare()`` 

53 yields a reusable resource. 

54 

55 A long ``rate_limit.cooldown_seconds`` stalls *all* callers during a 429 — 

56 tune it via ``config`` to trade upstream protection against latency. 

57 

58 Two opt-in knobs bound the request path under load (both off by default, so 

59 the default is pure backpressure — callers beyond ``max_workers`` wait): 

60 

61 - ``max_pending`` — an admission cap. With it set, at most 

62 ``max_workers + max_pending`` requests may be in flight (running *or* 

63 waiting on the semaphore); further submits are rejected *instantly* with a 

64 failed result instead of growing an unbounded waiter list. Bounds memory. 

65 - ``submit_timeout`` — a per-caller latency bound (seconds). An admitted 

66 request that hasn't completed within the budget (e.g. stuck behind a 

67 cooldown) is cancelled and returns a failed result. A firing timeout 

68 cancels an in-flight request that might have been about to succeed — the 

69 right trade for a web latency budget where the client is already gone. 

70 """

72 def __init__( 

73 self, 

74 strategy: LLMCallStrategy[TOutput], 

75 *, 

76 config: ProcessorConfig | None = None, 

77 error_classifier: Any = None, 

78 max_pending: int | None = None, 

79 submit_timeout: float | None = None, 

80 ) -> None: 

81 if max_pending is not None and max_pending < 0: 

82 raise ValueError(f"max_pending must be >= 0 (got {max_pending})") 

83 if submit_timeout is not None and submit_timeout <= 0: 

84 raise ValueError(f"submit_timeout must be > 0 (got {submit_timeout})") 

85 

86 cfg = config or ProcessorConfig(max_workers=5) 

87 # Always an int after ProcessorConfig.__post_init__ resolution. 

88 gateway_workers = cast(int, cfg.max_workers) 

89 warn_if_worker_capacity_exceeded( 

90 strategy=strategy, 

91 max_workers=gateway_workers, 

92 surface="LLMGateway", 

93 stacklevel=3, 

94 ) 

95 self._strategy = strategy 

96 self._host: ExecutorHost[Any, TOutput, Any] = ExecutorHost( 

97 cfg, strategy=strategy, error_classifier=error_classifier 

98 ) 

99 self._sem = asyncio.Semaphore(gateway_workers) 

100 self._seq = 0 

101 self._closed = False 

102 

103 # Admission cap: max requests in flight (running + waiting). None = off. 

104 self._max_inflight = None if max_pending is None else gateway_workers + max_pending 

105 self._inflight = 0 

106 self._submit_timeout = submit_timeout 

107 # Set whenever no request is in flight; aclose() waits on it to drain 

108 # already-admitted work before cleaning up the shared strategy. 

109 self._idle = asyncio.Event() 

110 self._idle.set() 

111 # The single drain+cleanup task; concurrent aclose() callers all await it 

112 # so every `await gw.aclose()` returns only once cleanup is complete. 

113 self._close_task: asyncio.Task[None] | None = None 

114 

115 async def __aenter__(self) -> LLMGateway[TOutput]: 

116 return self 

117 

118 async def __aexit__(self, *exc_info: object) -> None: 

119 await self.aclose()

121 async def submit(self, prompt: str, *, timeout: float | None = None) -> TOutput: 

122 """Submit one prompt and await its output, raising on failure. 

123 

124 Blocks on the semaphore when the pool is saturated (backpressure), unless 

125 ``max_pending`` is set (then an over-cap submit raises immediately). A 

126 rejected/timed-out submit raises :class:`LLMCallError`. ``timeout`` 

127 overrides the gateway's ``submit_timeout`` for this call. 

128 """ 

129 return unwrap_result(await self.submit_result(prompt, timeout=timeout)) 

130 

131 async def submit_result( 

132 self, prompt: str, *, timeout: float | None = None 

133 ) -> WorkItemResult[TOutput, Any]: 

134 """Like :meth:`submit` but returns the WorkItemResult instead of raising. 

135 

136 Gives access to ``token_usage`` / ``metadata`` and lets callers branch 

137 on ``.success`` without exception handling. A request rejected by the 

138 admission cap or cut off by the timeout comes back as a failed result 

139 (``success=False``) rather than raising. 

140 

141 ``timeout`` overrides the gateway's ``submit_timeout`` for this call. 

142 """ 

143 if self._closed: 

144 raise RuntimeError("LLMGateway is closed") 

145 

146 self._seq += 1 

147 item_id = f"req-{self._seq}" 

148 

149 # Admission cap. The check + increment below run with no `await` between 

150 # them, so in asyncio this is race-free and correctly counts waiters 

151 # toward the bound. 

152 if self._max_inflight is not None and self._inflight >= self._max_inflight: 

153 return WorkItemResult( 

154 item_id=item_id, success=False, error="gateway saturated", context=None 

155 ) 

156 

157 effective_timeout = self._submit_timeout if timeout is None else timeout 

158 work_item: LLMWorkItem[Any, TOutput, Any] = LLMWorkItem( 

159 item_id=item_id, strategy=self._strategy, prompt=prompt 

160 ) 

161 

162 self._inflight += 1 

163 self._idle.clear() 

164 try: 

165 run = self._run(work_item) 

166 if effective_timeout is None: 

167 return await run 

168 # Wrap the whole `async with self._sem: execute(...)` coroutine — NOT 

169 # `wait_for(sem.acquire())`, which leaks the permit. Cancellation here 

170 # unwinds through __aexit__, releasing the slot. 

171 try: 

172 return await asyncio.wait_for(run, effective_timeout) 

173 except (TimeoutError, asyncio.TimeoutError): 

174 return WorkItemResult( 

175 item_id=item_id, success=False, error="submit timed out", context=None 

176 ) 

177 finally: 

178 self._inflight -= 1 

179 if self._inflight == 0: 

180 self._idle.set() 

181 

182 async def _run(self, work_item: LLMWorkItem[Any, TOutput, Any]) -> WorkItemResult[TOutput, Any]: 

183 """Acquire a concurrency slot, run the item, release the slot.""" 

184 async with self._sem: 

185 return await self._host.executor.execute(work_item)

187 async def aclose(self) -> None: 

188 """Stop accepting work and run the strategy's cleanup(). Idempotent. 

189 

190 Marks the gateway closed (new submits raise immediately), then waits for 

191 already-admitted requests — running *or* still waiting on the semaphore — 

192 to drain before cleaning up the shared strategy, whose clients/caches may 

193 still be in use. Set ``submit_timeout`` to bound how long an admitted 

194 request can hold up shutdown (otherwise the drain waits indefinitely). 

195 

196 Concurrent callers share one drain+cleanup task, so *every* 

197 ``await gw.aclose()`` returns only once cleanup has actually completed — 

198 not just because another call set the closed flag first. The shared task 

199 is ``shield``-ed, so cancelling one caller's ``aclose()`` doesn't abort 

200 the drain/cleanup for the others. 

201 """ 

202 # Set synchronously so new submits are rejected immediately, even before 

203 # the cleanup task below is scheduled. 

204 self._closed = True 

205 if self._close_task is None: 

206 self._close_task = asyncio.ensure_future(self._drain_and_close()) 

207 # shield so a cancelled waiter doesn't cancel the shared drain/cleanup. 

208 await asyncio.shield(self._close_task) 

209 

210 async def _drain_and_close(self) -> None: 

211 # No new admissions once _closed is set, so _inflight only decreases. 

212 if self._inflight > 0: 

213 await self._idle.wait() 

214 await self._host.aclose()
